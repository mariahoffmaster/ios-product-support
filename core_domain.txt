namespace CreditLib {
    public static class Loader 
    {

        public class CRRequest
        {
            public int crrequest_id;
            public int? filenum;
            public int requesttype_id;
        }
        public static List<CRRequest> GetPendingRequests(string server) 
        {
            List<CRRequest> reqs = new List<CRRequest>();
            
            
            
            
            string sql = "select  crqueue_id,filenum,requesttype_id from vCRQueue c where c.processed = 0 and c.requesttype_id = 1 and (has_error = 0 or has_error is null) order by request_time asc";
            
            Query q = new Query(sql, server);
            
            DataTable t = q.ExecuteTable();
            foreach(DataRow dr in t.Rows)
            {
                var r = new CRRequest();
                r.crrequest_id = dr["crqueue_id"].AsInt().Value;
                r.filenum = dr["filenum"].AsInt();
                r.requesttype_id = dr["requesttype_id"].AsInt().Value;
                reqs.Add(r);
            }
            reqs.Shuffle();
            return reqs;
        }
    }
}
