        public void SyncAll(string server)
        {
            bool exists = this.CreditFileExists(server, this.creditfile_id);
            int load_id_tmp = Database.Common.GetNextId(server, "credit_load_id");
            if (!exists)
            {
                CreditLib.Load.CreateNewLoadId(server, load_id_tmp, this.pull_date, null, true);
                this.InsertCreditFile(server, this.creditfile_id, load_id_tmp);
             }
            Task a = Task.Run(() => { SyncTradeLines(server); });
            Task b = Task.Run(() => { SyncScores(server); });
            Task c = Task.Run(() => { SyncInquiries(server); });
            Task d = Task.Run(() => { SyncComments(server); });
            Task e = Task.Run(() => { SyncAlerts(server); });
            Task.WaitAll(a, b, c, d, e);
            CreditLib.Load.ProcessCreditLoadNEW(server, load_id_tmp, this.creditfile_id);
        }

		{
			CreditRunner crs = null;
			if (request.requesttype_id == 1)
			{

                bool needs_update = CreditLib.Queue.NeedsUpdate(server, request.crrequest_id, crs.cr.status, crs.cr.status_description, crs.cr.error);

        public static async Task<Dictionary<int, CreditRunner>> LoadRequestsBulk(string server, List<Loader.CRRequest> reqs, int threads,IETA eta)
        {
            
            Dictionary<int, CreditRunner> status = new Dictionary<int, CreditRunner>();
            List<Task> tasks = new List<Task>();
            var throttler = new SemaphoreSlim(threads, threads);
            foreach (var req in reqs)
            {
				await throttler.WaitAsync();
				tasks.Add(Task.Run(() =>
				{
					
					try
					{
						Console.WriteLine($"Starting New Task: {req.filenum}");
						
						LoadSingleCRSRequest(server, req);
						if (null != eta)
							eta.IncrementAndReport();

						Console.WriteLine($"Ending New Task: {req.filenum}");
					}
					finally { throttler.Release(); }
				}));
			}
            Task.WaitAll(tasks.ToArray());
            
            return status;
        }

declare @helper int
declare @months_back int = 2

if @before > @after
  begin
    set @helper = @after
    set @after = @before
    set @before = @helper
  end

drop table if exists #tmp
drop table if exists #summary

select 
  cl.asOf
  , ctls.creditfile_id
  , cat.[name] [account_type]
  , ctls.creditor_name 
  , sum(ctls.balance) [balance]
  , sum(ctls.high_credit_amount) [high_credit]
  , sum(ctls.monthly_payment) [payments]
  , sum(ctls.balance_past_due) [past_due]
  
  , cat.name [credit_account_type]
  , cat.ordering_default
  into #tmp
  from 
    CreditTradeLinesSimple ctls
    join CreditAccountTypes cat on cat.id = ctls.accounttype_id
    join CreditLoads cl on cl.id = ctls.load_id
    join CreditAccountStatus cas on cas.id = ctls.accountstatus_id
    WHERE 
      creditfile_id in (@before,@after)
      
      
      
      and (cas.charge_off = 0 or @ignore_charge_offs=0)
      and ctls.possible_dupe = 0
      and ctls.closed = 0
      
    group by
      cl.asOf
      , cat.name
      , cat.ordering_default
      , CTLS.creditor_name
      , ctls.creditfile_id
    order by
      cat.ordering_default ASC

        public static void WriteAllTax(string server, TaxHelper th)
        {
            int creditfile_id = th.full_call.DEAL_SETS.DEAL_SET.DEALS.DEAL.SERVICES.SERVICE.SERVICE_PRODUCT_FULFILLMENT.SERVICE_PRODUCT_FULFILLMENT_DETAIL.VendorOrderIdentifier;
            if (null != th.tax)
            {
                foreach (var tax in th.tax.EIVTAXTRANSCRIPT_V100)
                {
                    int taxtranscript_id = InsertTaxTranscript(server, creditfile_id, tax.TRACKINGNUMBER, tax.YEAR, Convert.ToDateTime(tax.TAXPERIODENDING));
                    foreach (var form in tax.FORM)
                    {
                        int taxform_id = WriteTaxForm(server, taxtranscript_id, form);
                        if (form.FORMTYPE == "1040" && null != form.FORM1040HEADER)
                        {
                            Write1040(server, taxform_id, form.FORM1040HEADER);
                        }
                        foreach (var section in form.SECTIONS)
                        {
                            int taxsection_id = WriteSection(server, taxform_id, section);
                            foreach (var vals in section.SECTIONFIELDS)
                            {
                                WriteSectionValues(server, taxsection_id, vals);
                            }
                        }
                    }
                }
            }
        
        }
