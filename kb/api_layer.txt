        public static void ReconcileTables()
        {
            string server = "UAM_REPORTING"; 
            string[] tablesToReconcile = new string[]
            {
        "Assets",
        "PropertyInfo",
        "Applicants",
        "AssetApplicantMap",
        "Underwriting",
        "Debt"
            };

            List<string> differences = new List<string>();
            differences.Add("Table,Column,AssetId,GraphqlValue,NonGraphqlValue");

            foreach (string table in tablesToReconcile)
            {
                Console.WriteLine($"Reconciling table {table}...");
                string graphqlTable = table + "_graphql";
                string nonGraphqlTable = table;

                string fetchGraphqlDataSql = $"SELECT * FROM {graphqlTable};";

                string fetchNonGraphqlDataSql = $"SELECT * FROM {nonGraphqlTable} WHERE asset_id = @key;";
                if (table.Equals("Assets"))
                {
                    fetchNonGraphqlDataSql = $"SELECT * FROM {nonGraphqlTable} WHERE ID = @key;";
                }
                else if (table.Equals("Applicants"))
                {
                    fetchNonGraphqlDataSql = $"SELECT * FROM {nonGraphqlTable} WHERE email = @key;";
                }
                else if (table.Equals("AssetApplicantMap"))
                {
                    fetchNonGraphqlDataSql = $"SELECT * FROM {nonGraphqlTable} WHERE asset_id = @asset_id AND applicanttype_id = @applicanttype_id;";
                }
                else if (table.Equals("Debt"))
                {
                    fetchNonGraphqlDataSql = $"SELECT * FROM {nonGraphqlTable} WHERE asset_id = @asset_id AND lien_position = @lien_position;";
                }

                Query fetchGraphqlDataQuery = new Query(fetchGraphqlDataSql, server);
                DataTable graphqlData = fetchGraphqlDataQuery.ExecuteTable();

                foreach (DataRow graphqlRow in graphqlData.Rows)
                {
                    string key = "";
                    Query fetchNonGraphqlDataQuery = new Query(fetchNonGraphqlDataSql, server);
                    if (table.Equals("Assets"))
                    {
                        key = graphqlRow["id"].ToString();
                        fetchNonGraphqlDataQuery.AddParameter("@key", key);
                    }
                    else if (table.Equals("Applicants"))
                    {
                        key = graphqlRow["email"].ToString();
                        fetchNonGraphqlDataQuery.AddParameter("@key", key);
                    }
                    else if (table.Equals("AssetApplicantMap"))
                    {
                        string assetId = graphqlRow["asset_id"].ToString();
                        string applicantTypeId = graphqlRow["applicanttype_id"].ToString();
                        fetchNonGraphqlDataQuery.AddParameter("@asset_id", assetId);
                        fetchNonGraphqlDataQuery.AddParameter("@applicanttype_id", applicantTypeId);
                        key = $"{assetId}-{applicantTypeId}";
                    }
                    else if (table.Equals("Debt"))
                    {
                        string assetId = graphqlRow["asset_id"].ToString();
                        string lienPosition = graphqlRow["lien_position"].ToString();
                        fetchNonGraphqlDataQuery.AddParameter("@asset_id", assetId);
                        fetchNonGraphqlDataQuery.AddParameter("@lien_position", lienPosition);
                        key = $"{assetId}-{lienPosition}";
                    }
                    else
                    {
                        key = graphqlRow["asset_id"].ToString();
                        fetchNonGraphqlDataQuery.AddParameter("@key", key);
                    }

                    DataTable nonGraphqlData = fetchNonGraphqlDataQuery.ExecuteTable();

                    if (nonGraphqlData.Rows.Count > 0)
                    {
                        DataRow nonGraphqlRow = nonGraphqlData.Rows[0];
                        foreach (DataColumn column in graphqlData.Columns)
                        {
                            string columnName = column.ColumnName;
                            string graphqlValue = graphqlRow[columnName]?.ToString() ?? string.Empty;
                            string nonGraphqlValue = nonGraphqlRow[columnName]?.ToString() ?? string.Empty;

                            if (columnName.Equals("update_time") || columnName.Equals("update_by") || columnName.Equals("id"))
                            {
                                continue;
                            }

                            if (graphqlValue != nonGraphqlValue && !graphqlValue.Equals(nonGraphqlValue))
                            {
                                differences.Add($"{table},{columnName},{key},\"{graphqlValue.Replace("\"", "\"\"")}\",\"{nonGraphqlValue.Replace("\"", "\"\"")}\"");

                                
                                string insertDifferenceSql = @"
                        INSERT INTO ReconciliationDifferences (TableName, ColumnName, AssetId, GraphqlValue, NonGraphqlValue)
                        VALUES (@TableName, @ColumnName, @AssetId, @GraphqlValue, @NonGraphqlValue);";

                                Query insertDifferenceQuery = new Query(insertDifferenceSql, server);
                                insertDifferenceQuery.AddParameter("@TableName", table);
                                insertDifferenceQuery.AddParameter("@ColumnName", columnName);
                                insertDifferenceQuery.AddParameter("@AssetId", key);
                                insertDifferenceQuery.AddParameter("@GraphqlValue", graphqlValue);
                                insertDifferenceQuery.AddParameter("@NonGraphqlValue", nonGraphqlValue);
                                insertDifferenceQuery.Execute();
                            }
                        }
                    }
                    else
                    {
                        differences.Add($"{table},ALL,{key},Row exists,Row does not exist");

                        
                        string insertDifferenceSql = @"
                INSERT INTO ReconciliationDifferences (TableName, ColumnName, AssetId, GraphqlValue, NonGraphqlValue)
                VALUES (@TableName, 'ALL', @AssetId, 'Row exists', 'Row does not exist');";

                        Query insertDifferenceQuery = new Query(insertDifferenceSql, server);
                        insertDifferenceQuery.AddParameter("@TableName", table);
                        insertDifferenceQuery.AddParameter("@AssetId", key);
                        insertDifferenceQuery.Execute();
                    }
                }
            }

            
            string csvFilePath = "C:\\git\\vs\\GraphQL-V2\\GraphQL-V2\\reconciliation_differences.csv";
            File.WriteAllLines(csvFilePath, differences);
            Console.WriteLine($"Reconciliation completed. Differences saved to {csvFilePath}");
        }

        public static (int Beds, decimal Baths, int SqFt)? GetBedsBathsSqFtFromJson(string valuationsJson)
        {
            
            var valuations = JsonConvert.DeserializeObject<List<dynamic>>(valuationsJson);

            if (valuations == null || !valuations.Any())
                return null;

            
            foreach (var valuation in valuations)
            {
                if (valuation.propertyBeds != null && valuation.propertyBaths != null && valuation.propertySquareFeet != null)
                {
                    int beds = (int)valuation.propertyBeds;
                    decimal baths = decimal.Parse((string)valuation.propertyBaths);
                    decimal squareFeet = decimal.Parse((string)valuation.propertySquareFeet);
                    int sqft = (int)Math.Round(squareFeet);
                    return (beds, baths, sqft);
                }
            }

            return null; 
        }

                return null;

            
            foreach (var valuation in valuations)
            {
                if (valuation.propertyBeds != null && valuation.propertyBaths != null && valuation.propertySquareFeet != null)
                {
                    int beds = (int)valuation.propertyBeds;
                    decimal baths = decimal.Parse((string)valuation.propertyBaths);
                    decimal squareFeet = decimal.Parse((string)valuation.propertySquareFeet);
                    int sqft = (int)Math.Round(squareFeet);
                    return (beds, baths, sqft);
                }
            }

            return null; 
        }

        public static async Task<string> GetAccessTokenAsync(string userName, string password)
        {
            using (var client = new HttpClient())
            {
                var url = "https:
                var payload = new
                {
                    username = userName,
                    password = password
                };
                var content = new StringContent(JsonConvert.SerializeObject(payload), Encoding.UTF8, "application/json");

                var response = await client.PostAsync(url, content);
                response.EnsureSuccessStatusCode();

                var responseContent = await response.Content.ReadAsStringAsync();
                var tokenResponse = JsonConvert.DeserializeObject<TokenResponse>(responseContent);

                return tokenResponse.Access;
            }
        }

        {
            string server = "UAM_REPORTING"; 
            string[] tablesToReconcile = new string[]
            {
        "Assets",
        "PropertyInfo",
        "Applicants",
        "AssetApplicantMap",
        "Underwriting",
        "Debt"
            };

            List<string> differences = new List<string>();
            differences.Add("Table,Column,AssetId,GraphqlValue,NonGraphqlValue");

            foreach (string table in tablesToReconcile)
            {
                Console.WriteLine($"Reconciling table {table}...");
                string graphqlTable = table + "_graphql";
                string nonGraphqlTable = table;

                string fetchGraphqlDataSql = $"SELECT * FROM {graphqlTable};";

                string fetchNonGraphqlDataSql = $"SELECT * FROM {nonGraphqlTable} WHERE asset_id = @key;";
                if (table.Equals("Assets"))
                {
                    fetchNonGraphqlDataSql = $"SELECT * FROM {nonGraphqlTable} WHERE ID = @key;";
                }
                else if (table.Equals("Applicants"))
                {
                    fetchNonGraphqlDataSql = $"SELECT * FROM {nonGraphqlTable} WHERE email = @key;";
                }
                Query fetchGraphqlDataQuery = new Query(fetchGraphqlDataSql, server);
                DataTable graphqlData = fetchGraphqlDataQuery.ExecuteTable();

                foreach (DataRow graphqlRow in graphqlData.Rows)
                {
                    string key = "";
                    Query fetchNonGraphqlDataQuery = new Query(fetchNonGraphqlDataSql, server);
                    if (table.Equals("Assets"))
                    {
                        key = graphqlRow["id"].ToString();
                    }
                    else if (table.Equals("Applicants"))
                    {
                        key = graphqlRow["email"].ToString();
                    }
                    else
                    {
                        key = graphqlRow["asset_id"].ToString();
                    }
                    fetchNonGraphqlDataQuery.AddParameter("@key", key);
                    DataTable nonGraphqlData = fetchNonGraphqlDataQuery.ExecuteTable();

                    if (nonGraphqlData.Rows.Count > 0)
                    {
                        DataRow nonGraphqlRow = nonGraphqlData.Rows[0];
                        foreach (DataColumn column in graphqlData.Columns)
                        {
                            string columnName = column.ColumnName;
                            string graphqlValue = graphqlRow[columnName]?.ToString() ?? string.Empty;
                            string nonGraphqlValue = nonGraphqlRow[columnName]?.ToString() ?? string.Empty;

                            if (columnName.Equals("update_time") || columnName.Equals("update_by"))
                            {
                                continue;
                            }

                            if (graphqlValue != nonGraphqlValue && !graphqlValue.Equals(nonGraphqlValue))
                            {
                                differences.Add($"{table},{columnName},{key},\"{graphqlValue.Replace("\"", "\"\"")}\",\"{nonGraphqlValue.Replace("\"", "\"\"")}\"");
                            }
                        }
                    }
                    else
                    {
                        differences.Add($"{table},ALL,{key},Row exists,Row does not exist");
                    }
                }
            }

            
            string csvFilePath = "reconciliation_differences.csv";
            File.WriteAllLines(csvFilePath, differences);
            Console.WriteLine($"Reconciliation completed. Differences saved to {csvFilePath}");
        }

namespace GraphQL
{
    public class Recon
    {
        private readonly string _server;

        public Recon(string server)
        {
            _server = server;
        }

        public void CompareTables(string table1, string table2, string outputCsvPath)
        {
            DataTable dt1 = LoadTable(table1);
            DataTable dt2 = LoadTable(table2);

            var differences = GetDifferences(dt1, dt2);
            WriteDifferencesToCsv(differences, outputCsvPath);
        }

        private DataTable LoadTable(string tableName)
        {
            string sql = $"SELECT * FROM {tableName}";
            Query q = new Query(sql, _server);
            return q.ExecuteTable();
        }

        private List<Dictionary<string, object>> GetDifferences(DataTable dt1, DataTable dt2)
        {
            var differences = new List<Dictionary<string, object>>();

            var dt1Dict = dt1.AsEnumerable().Select(row => row.ItemArray.Select(item => item?.ToString() ?? "").ToArray()).ToList();
            var dt2Dict = dt2.AsEnumerable().Select(row => row.ItemArray.Select(item => item?.ToString() ?? "").ToArray()).ToList();

            foreach (var row1 in dt1Dict)
            {
                if (!dt2Dict.Any(row2 => row2.SequenceEqual(row1)))
                {
                    var diff = new Dictionary<string, object> { { "Status", "Deleted" } };
                    for (int i = 0; i < dt1.Columns.Count; i++)
                    {
                        diff[dt1.Columns[i].ColumnName] = row1[i];
                    }
                    differences.Add(diff);
                }
            }

            foreach (var row2 in dt2Dict)
            {
                if (!dt1Dict.Any(row1 => row1.SequenceEqual(row2)))
                {
                    var diff = new Dictionary<string, object> { { "Status", "Added" } };
                    for (int i = 0; i < dt2.Columns.Count; i++)
                    {
                        diff[dt2.Columns[i].ColumnName] = row2[i];
                    }
                    differences.Add(diff);
                }
            }

            return differences;
        }

        private void WriteDifferencesToCsv(List<Dictionary<string, object>> differences, string outputCsvPath)
        {
            if (differences.Count == 0)
            {
                Console.WriteLine("No differences found.");
                return;
            }

            using (var writer = new StreamWriter(outputCsvPath, false, Encoding.UTF8))
            using (var csv = new CsvWriter(writer, new CsvConfiguration(CultureInfo.InvariantCulture)))
            {
                var headers = differences.SelectMany(d => d.Keys).Distinct().ToList();
                csv.WriteRecords(differences.Select(d => headers.ToDictionary(h => h, h => d.ContainsKey(h) ? d[h] : "")));
            }

            Console.WriteLine($"CSV saved to {outputCsvPath}");
        }
    }

            prev.remaining_bom_pct = 1.0;

            prev.remaining_eom_pct = 1.0;

            for (int month = 1; month != a.term_months+1; month++)
            {
                DataRow row = flows.Rows.Add();

                CashflowSingle cf = new CashflowSingle();

                cf.cpr = pi.cpr;
                cf.cdr = pi.cdr;
                cf.sev = pi.sev;
                cf.hpi = pi.hpi;
                DateTime cashflow_month = a.effective_date.AddMonths(month).SetDay(1);


                cf.cpr_smm = 1.0 - (Math.Pow(1.0-cf.cpr/100.0, 1.0 / 12.0));
                cf.cdr_smm = 1.0 - (Math.Pow(1.0 - cf.cdr / 100.0, 1.0 / 12.0));

                if (month == a.term_months)
                    cf.cpr_smm = 1;

                cf.remaining_bom_pct = prev.remaining_eom_pct;
                cf.remaining_after_default_pct = cf.remaining_bom_pct * (1.0 - cf.cdr_smm);
                cf.remaining_eom_pct = cf.remaining_bom_pct * (1 - (cf.cpr_smm + cf.cdr_smm));


                cf.current_home_value = Math.Pow(1.0 + cf.hpi, 1.0 / 12.0) * prev.current_home_value;
                
                int days = (cashflow_month-a.effective_date.Date).Days;
                cf.maximum_unlock_share = a.cost_limit_basis * Math.Pow(1.0 + a.cost_limit, days / 365.0);
      
                double unlock_pct_bom = a.unlock_pct * cf.current_home_value;
                cf.unlock_share_capped = Math.Min(unlock_pct_bom, cf.maximum_unlock_share);

                cf.prepay_month = cf.cpr_smm * cf.unlock_share_capped * cf.remaining_after_default_pct;
                cf.default_month = cf.cdr_smm * cf.unlock_share_capped * cf.remaining_bom_pct;
                cf.recovery_month = cf.default_month * (1.0 - cf.sev /100.0);


                cf.remaining_bom_inv_pmt = cf.remaining_bom_pct * a.investment_payment;
                cf.remaining_eom_inv_pmt = cf.remaining_eom_pct * a.investment_payment;
                cf.month_num = month;           
                cf.total_contract_cash_month = cf.prepay_month + cf.recovery_month;
                cf.month = cashflow_month;
                cf.wal_calc = cf.remaining_bom_pct - cf.remaining_eom_pct;


                row["month"] = month;
                row["prepay"] = cf.prepay_month;
                row["default"] = cf.default_month;
                row["recovery"] = cf.recovery_month;
                row["cashflow"] = cf.total_contract_cash_month;
                total_cashflow += cf.total_contract_cash_month;
                prev = cf;
                _flows.Add(cf.month, cf);


            }
            double total_cash = _flows.Sum(n => n.Value.total_contract_cash_month);
            
            cfs.LoadFlows(_flows);

        public static double BXIRR(double[] cashflows, DateTime[] dates, UtilsLib.Dates.DayCount dc)
        {
            if (cashflows.Length != dates.Length)
                return -9999999999;



            int max_runs = 150000;
            double precision = .01;
            double step_orig = .001;
            double step = step_orig;
            double high_level_precision = 1000;
            double price_paid = -cashflows[0];

            double last_yield = 0.01;
            Nullable<double> last_npv;
            for (int i = 0; i != max_runs; i++)
            {
                double npv = BXNPV(last_yield, cashflows, dates, dc);
                double delta = npv;
                if (-precision < delta && delta < precision)
                    
                    return last_yield;

                if (-high_level_precision < delta && delta < high_level_precision)
                {
                    step_orig = .0001;
                }


                if (delta > 0)
                    last_yield = last_yield * (1 + step_orig);
                else
                    last_yield = last_yield * (1 - step_orig);
                
                
                
                
                
                
                
                

            }
            return -9999999999999999;
        }

        public static double BXNPV(double discount_rate, double[] cashflows, DateTime[] dates, UtilsLib.Dates.DayCount dc)
        {
            if (discount_rate < 1)
                discount_rate = discount_rate * 100;
            Nullable<DateTime> previous_date = null;
            Nullable<DateTime> settle_date = null;
            Nullable<double> discount_factor = 0;
            
            double discount_factor_new = 1;
            double discount_factor_new_prev = 1;
            double total_npv = 0;
            for (int i = 0; i != cashflows.Length; i++)
            {
                double cashflow = cashflows[i];
                
                DateTime date = dates[i];
                if (i == 0)
                {
                    previous_date = dates[0];
                    discount_factor = 1;
                    settle_date = dates[0];
                    total_npv += cashflow;
                    continue;
                }
                double days_since = (date - previous_date.Value).TotalDays;
                double period_time_change = days_since / 360.0;
                double time_in_years = UtilsLib.Dates.Yearfrac(previous_date.Value, date, dc);
                discount_factor = discount_factor_new_prev / (1 + ((days_since) / (360.0) * (discount_rate / 100.0)));
                discount_factor = discount_factor_new_prev / (1 + (time_in_years * (discount_rate / 100.0)));
                


                
                double npv = discount_factor.Value * cashflow;
                total_npv += npv;
                previous_date = date;
                discount_factor_new_prev = discount_factor.Value;
            }
            return total_npv;
        }

    {



        public XIRRCalculator(double lowRate, double highRate, List<CashFlowDates> cashFlow)
        {
            var cashFlowYears = ToFractionOfYears(cashFlow);
            Initialize(lowRate, highRate, cashFlowYears);

        }
        private XIRRCalculator(double lowRate, double highRate, List<CashFlowFractionOfYear> cashFlow)
        {
            Initialize(lowRate, highRate, cashFlow);
        }
        private void Initialize(double lowRate, double highRate, List<CashFlowFractionOfYear> cashFlow)
        {
            LowRate = lowRate;
            HighRate = highRate;
            CashFlow = cashFlow;
            LowResult = CalcEquation(CashFlow, LowRate);
            HighResult = CalcEquation(CashFlow, HighRate);
        }
        private double LowRate { get; set; }
        private double HighRate { get; set; }
        private double LowResult { get; set; }
        private double HighResult { get; set; }
        private List<CashFlowFractionOfYear> CashFlow { get; set; }

        private double CalcEquation(List<CashFlowFractionOfYear> cashflows, double interestRate)
        {
            return cashflows.Select(x => (x.Amount / (Math.Pow((1 + interestRate), x.Years)))).Sum(x => x);
        }
        private static List<CashFlowFractionOfYear> ToFractionOfYears(List<CashFlowDates> cashflows)
        {
            var firstDate = cashflows.Min(x => x.Date);
            return cashflows
                .Select(x => new CashFlowFractionOfYear(x.Amount, ((double)x.Date.Subtract(firstDate).Days) / 365))
                .ToList();
        }

        public double? Calculate(double precision, int decimals)
        {
            if (Math.Sign(LowResult) == Math.Sign(HighResult))
            {
                return null;
                
            }

            var middleRate = (LowRate + HighRate) / 2;
            var middleResult = CalcEquation(CashFlow, middleRate);
            if (Math.Sign(middleResult) == Math.Sign(LowResult))
            {
                LowRate = middleRate;
                LowResult = middleResult;
            }
            else
            {
                HighRate = middleRate;
                HighResult = middleResult;
            }
            if (Math.Abs(middleResult) > precision)
            {
                return Calculate(precision, decimals);
            }
            else
            {
                return Math.Round((HighRate + LowRate) / 2, decimals);
            }

        }

        public class CashFlowFractionOfYear
        {

            {
                this.Amount = Amount;
                this.Years = Years;
            }

                    d.Amount = System.Convert.ToDouble(row[flow_col]);

            }
