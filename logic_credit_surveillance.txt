        public void SendMessage(RECIPIENT chan, string message, SEVERITY severity = SEVERITY.info, EMOJI emoji = EMOJI.None, List<TAGGED_RECIPIENT> userTags = null)
        {
            _client = new SlackClient(hooks[chan]);

            if (emoji != EMOJI.None)
                message = $"{GetEmoji(emoji)} {message}";

            if (userTags != null && userTags.Count > 0)
                foreach (var userTag in userTags)
                    message = $"{message} <@{GetTaggedRecipient(userTag, server)}>";

            var slackMessage = new SlackMessage();

            if (severity == SEVERITY.info)
                slackMessage.Text = message;
            else
            {
                slackMessage.Attachments = new List<SlackAttachment>
                {
                    new SlackAttachment
                    {
                        Text = message,
                        Color = GetSeverityColor(severity)
                    }
                };
            }

            try { _client.Post(slackMessage); }
            catch (Exception e) { Console.WriteLine(e.Message); }
        }

                { EMOJI.MoneyWithWings, ":money_with_wings:" },
                { EMOJI.Dollar, ":dollar:" },
                { EMOJI.Unlock, ":unlock:" },
                { EMOJI.Bank, ":bank:" },
                { EMOJI.HeavyDollarSign, ":heavy_dollar_sign:" },
                { EMOJI.CurrencyExchange, ":currency_exchange:" }
            };

            return emojiDict.ContainsKey(emoji) ? emojiDict[emoji] : Emoji.Moneybag;
        }

        private string GetSeverityColor(SEVERITY severity)
        {
            var colorDict = new Dictionary<SEVERITY, string>
            {
                { SEVERITY.severe, "danger" },    
                { SEVERITY.warning, "warning" },  
                { SEVERITY.success, "#36a64f" },     
                { SEVERITY.insight, "#00BFFF" }   
            };

            return colorDict.ContainsKey(severity) ? colorDict[severity] : "#36a64f";
        }

            Query q = new Query(sql, server);
            q.AddParameter("@email", email);
            return q.ExecuteSingle<string>("slack_tag");
        }

namespace DocuSign.Lib
{
    public class DSHelper
    {
        public static string PrepareFullPrivateKeyFilePath(string fileName)
        {
            const string DefaultRSAPrivateKeyFileName = "private.txt";

            var fileNameOnly = Path.GetFileName(fileName);
            var filePath = Path.GetDirectoryName(fileName);

            if (string.IsNullOrEmpty(fileNameOnly))
            {
                fileNameOnly = DefaultRSAPrivateKeyFileName;
            }

            if (string.IsNullOrEmpty(filePath))
            {
                filePath = Directory.GetCurrentDirectory();
            }

            return Path.Combine(filePath, fileNameOnly);
        }

        public static byte[] ReadFileContent(string path)
        {
            return File.ReadAllBytes(path);
        }
    }
}

namespace DocuSign.Lib.Dtos
{
    using eSigner = eSign.Model.Signer;

    public abstract class EnvelopeRecipient
    {
        protected internal Recipients SetRecipient(IEnumerable<DocuSignLib.Dtos.Signer> signers)
        {
            var recipients = new Recipients() { Signers = new List<eSigner>() };
            foreach (var signer in signers)
            {
                recipients.Signers.Add(SetSigner(signer.Email, signer.Name, signer.SignerId, signer.RoutingOrder));
            }
            return recipients;
        }

        protected internal eSigner SetSigner(string signerEmail, string signerName, string signerId, string routingOrder)
        {
            return new eSigner
            {
                Email = signerEmail,
                Name = signerName,
                RecipientId = signerId,
                RoutingOrder = routingOrder,
                RoleName = $"{ RoleTypes.Signer.ToString()}",

                Tabs = new Tabs
                {
                    SignHereTabs = new List<SignHere>
                    {
                        new SignHere
                        {
                            AnchorString = $"s{routingOrder}",
                            RecipientId = signerId,
                            TabType = "SignHere",
                            AnchorXOffset = "1",
                            AnchorYOffset = "0"
                        }
                    }
                }
            };
        }
    }
}

namespace DocuSign.Lib.Implementations
{
    public class EnvelopeProvider : EnvelopeRecipient, IEnvelopeProvider
    {
        private readonly EnvelopeDefinition _envDef;
        public EnvelopeProvider()
        {
            _envDef = new EnvelopeDefinition();
        }

        public Response CreateAndSend(Dtos.Envelope envelope, Account account)
        {
            var customApi = CustomApi.Instance.Envelopes(account.AccessToken, account.BasePath);

            _envDef.Recipients = SetRecipient(envelope.Signers);

            _envDef.Documents = new List<eSign.Model.Document>();
              foreach(var doc in envelope.Documents)
            {
                _envDef.Documents.Add(new eSign.Model.Document
                {
                    DocumentBase64 = doc.Base64String,
                    FileExtension = doc.Extension,
                    DocumentId = doc.Id,
                    Name = doc.Name
                });
            }
            
            

            _envDef.Status = "Created";
            _envDef.EmailBlurb = envelope.Email.EmailBody;
            _envDef.EmailSubject = envelope.Email.EmailSubject;
            var created =  customApi.CreateEnvelope(account.AccountId, _envDef);
            if (!string.IsNullOrWhiteSpace(created.EnvelopeId))
            {
                var envelopeInfo = customApi.GetEnvelope(account.AccountId, created.EnvelopeId);
                envelopeInfo.Status = "Sent";
                envelopeInfo.PurgeState = "";

                var result = customApi.Update(account.AccountId, created.EnvelopeId, envelopeInfo);

                return new Response { Message = "Envelope sent successfully", StatusCode = 200, StatusDesc = "OK" };
            }
            return new Response { Message = "Envelope could not sent", StatusCode = 404, StatusDesc = "Not Found" };
        }
    }
}

using static DocuSign.eSign.Client.Auth.OAuth;

namespace DocuSign.Lib.Implementations
{
    public class RequestItemsService : IRequestItemsService
    {
        private OAuthToken _authToken;

        protected static DocuSignClient _apiClient { get; private set; }
        private static UserInfo.Account _account { get; set; }

        public RequestItemsService()
        {
            Status = "sent";
            _apiClient = new DocuSignClient();
        }
        public void UpdateUserFromJWT()
        {
            var scopes = new List<string>
                {
                    "signature",
                    "impersonation"
                };

            _authToken = _apiClient.RequestJWTUserToken(
                ConfigurationManager.AppSettings["ClientId"],
                ConfigurationManager.AppSettings["ImpersonatedUserId"],
                ConfigurationManager.AppSettings["AuthServer"],
                DSHelper.ReadFileContent(DSHelper.PrepareFullPrivateKeyFilePath(
                    ConfigurationManager.AppSettings["PrivateKeyFile"])), 1, scopes);

            _account = _account == null ? GetAccountInfo(_authToken) : _account;

            User = new Account
            {
                Name = _account.AccountName,
                AccessToken = _authToken.access_token,
                ExpireIn = DateTime.Now.AddSeconds(_authToken.expires_in.Value),
                AccountId = _account.AccountId,
                BasePath = $"{_account.BaseUri}/restapi",

            };
        }

        public Account User { get; set; }
        public void Logout()
        {
            this._authToken = null;
        }

        public bool CheckTokenExpiration(int bufferMin = 60)
        {
            bool isJWTGrantAuthenticated = User?.AccessToken != null
                    && (DateTime.Now.Subtract(TimeSpan.FromMinutes(bufferMin)) < User.ExpireIn.Value);

            return  isJWTGrantAuthenticated;
        }

        public string Status { get; set; }
        public string XDocuSignSigntaure1 { get; set; }

        private UserInfo.Account GetAccountInfo(OAuthToken authToken)
        {
            _apiClient.SetOAuthBasePath(ConfigurationManager.AppSettings["AuthServer"]);

            UserInfo userInfo = _apiClient.GetUserInfo(authToken.access_token);
            UserInfo.Account acct = userInfo.Accounts.FirstOrDefault(accnt => accnt.IsDefault.Equals("true", StringComparison.OrdinalIgnoreCase));
          
            if (acct == null)
            {
                throw new Exception("The user does not have access to account");
            }

            return acct;
        }
    }
}

        public static byte[] CompressDataSet(DataSet ds)
        {
             System.IO.MemoryStream objStream = new MemoryStream();
             try
             {  
                 ds.RemotingFormat = SerializationFormat.Binary;
                 BinaryFormatter bf = new BinaryFormatter();
                 MemoryStream ms = new MemoryStream();

                 bf.Serialize(ms, ds);
                 byte[] inbyt = ms.ToArray();

                 System.IO.Compression.DeflateStream objZS = new System.IO.Compression.DeflateStream(objStream, System.IO.Compression.CompressionMode.Compress);

                objZS.Write(inbyt, 0, inbyt.Length);              
                objZS.Flush();
                objZS.Close();
              }
              catch (Exception ex)
              {               
                        throw ex;
              }

                return objStream.ToArray();
        }


        public static byte[] ReadFullStream(Stream stream)
        {
            byte[] buffer = new byte[32768];
            using (MemoryStream ms = new MemoryStream())
            {
                while (true)
                {
                    int read = stream.Read(buffer, 0, buffer.Length);
                    if (read <= 0)
                        return ms.ToArray();
                    ms.Write(buffer, 0, read);
                }
            }
        }

        public static DataSet DecompressDataSet(byte[] bytDs)
        {
            DataSet outDs = new DataSet();

            MemoryStream inMs = new MemoryStream(bytDs);
            inMs.Seek(0, 0);
            DeflateStream zipStream = new DeflateStream(inMs, CompressionMode.Decompress, true);

            byte[] outByt = ReadFullStream(zipStream);
            zipStream.Flush();
            zipStream.Close();

            MemoryStream outMs = new MemoryStream(outByt);
            outMs.Seek(0, 0);
            outDs.RemotingFormat = SerializationFormat.Binary;
            BinaryFormatter bf = new BinaryFormatter();
            

            outDs = (DataSet)bf.Deserialize(outMs, null);

            return outDs;
        }

        public DataTable GetData(List<string> bonds, List<string> fields,Dictionary<string,string> overrides = null)

                    bonds[i] = bonds[i];

                    bonds[i] = bonds[i];

                    bonds[i] = bonds[i];
                else if (bonds[i].ToUpper().Contains("INDEX"))
                    bonds[i] = bonds[i];
                else if (bonds[i].ToUpper().Contains("CURNCY"))
                    bonds[i] = bonds[i];
                else
                {
                    
                    bonds[i] = bonds[i] + " MTGE";
                }
            }
            if (!fields.Contains("SECURITY_DES") && bonds.Count != 1)
                fields.Add("SECURITY_DES");

                return CopyDataTableCells(tbl);

                cache_db = GetDbCache(CommonInterfaces.Constants.Database.SQL_SERVER,hash);

            if (null != cache_db)
            {
                if (!_cache.ContainsKey(hash))
                {
                    _cache.Add(hash, CopyDataTableCells(cache_db));

                return CopyDataTableCells(cache_db);
            }
            DataTable results = null;

             sendRefDataRequest(_session, bonds, fields, overrides);

                
             results = eventLoop(_session);

            
            if(results.Rows.Count == 2 && bonds.Count == 1)

        private DataTable eventLoop(Session session)
        {
            DataTable results = null;
            bool done = false;
            while (!done)
            {
                Event eventObj = session.NextEvent();
               if (eventObj.Type == Event.EventType.PARTIAL_RESPONSE)
                {
                    System.Console.WriteLine("Processing Partial Response");
                    processResponseEvent(eventObj);
                }
                else if (eventObj.Type == Event.EventType.RESPONSE)
                {
                    System.Console.WriteLine("Processing Response");
                    System.Diagnostics.Debug.WriteLine("Processing Response");
                    
                    results = processResponseEvent(eventObj);
                    done = true;
                }
                else
                {
                    foreach (Message msg in eventObj.GetMessages())
                    {
                        System.Console.WriteLine(msg.AsElement);
                        if (eventObj.Type == Event.EventType.SESSION_STATUS)
                        {
                            if (msg.MessageType.Equals("SessionTerminated"))
                            {
                                done = true;
                            }
                        }
                    }
                }
            }

            return results;
        }

                                results.Columns.Add(field.Name.ToString(), t);

                            if (!results.RowExists("SECURITY_DES", ticker))
                            {
                                DataRow row = results.Rows.Add();
                                row["SECURITY_DES"] = ticker;
                            }
                            DataRow asset_row = results.GetSingleRow("SECURITY_DES", ticker);
                            if (!field.IsNull)
                            {
                                object o = ConvertBBGData(field);
                                if (t == typeof(DataTable))
                                {
                                    o = ProcessSequence(field);
                                }

                                asset_row[field_name] = System.Convert.ChangeType(o, t);
                            }
                            
                           
                        }
                    }
                    else 
                        System.Console.WriteLine("No fields");

                }
            }
            return results;
        
        }

        public static object ConvertBBGData(Element e)
        {
            Type t = ConvertBBGType(e.Datatype);
            if (t == typeof(DataTable))
                return e;
            object o = e.GetValue();

            if (t == typeof(DateTime))
                o = System.Convert.ToDateTime(e.GetValueAsDatetime().ToString());
            return o;
        }


        public static DataTable ProcessSequence(Element refBulkField)
        {

            DataTable results = new DataTable();
            results.TableName = refBulkField.Name.ToString();
            
            int numofBulkValues = refBulkField.NumValues;
            for (int bvCtr = 0; bvCtr < numofBulkValues; bvCtr++)
            {
                Element bulkElement = refBulkField.GetValueAsElement(bvCtr);
                
                int numofBulkElements = bulkElement.NumElements;
                
                DataRow row = results.Rows.Add();
                for (int beCtr = 0; beCtr < numofBulkElements; beCtr++)
                {
                    Element elem = bulkElement.GetElement(beCtr);
                    string col = elem.Name.ToString().ToString().ToUpper();
                    Type t = ConvertBBGType(elem.Datatype);
                    if (!results.Columns.Contains(col))
                    {
                        results.Columns.Add(col, t);
                    }
                    row[col] = System.Convert.ChangeType(elem.GetValueAsString(),t);
                    
                }
            }
            return results;
        }

    }

        private DataTable ReadXlsxToDataTable(string filePath)
        {

            try
            {
                _xlh = new XLHelper(null);
                _xlh.CreateNewApp(false);

                if (_writer != null)
                    _writer("Opening xlsx file...");

                _xlh.OpenSheet(filePath, true);

                string worksheet = _xlh.ActiveWorksheetName();
                string usedRange = _xlh.UsedRangeAddressFull(worksheet);

                dt = _xlh.GetRangeDataTable(worksheet, usedRange);

                if (dt == null)
                    dt = new DataTable();
            }
            finally
            {
                if (_xlh != null)
                {
                    try { _xlh.Quit(5000); }
                    catch (Exception ex)
                    {
                        log.Error("Error closing Excel application during read", ex);
                    }
                    _xlh = null;
                }
            }

            return dt;
        }

            {

                xlh.ToggleDisplayAlerts(false);
                xlh.ToggleCalcs(XLHelper.XLCalcMode.MANUAL);

                string worksheet = xlh.ActiveWorksheetName();
                string startRange = "'" + worksheet + "'!A1";

                xlh.WriteRange(startRange, dt, true);
                xlh.SaveActiveWorkbookAs(filePath, true);
            }
            finally
            {
                if (xlh != null)
                {
                    try { xlh.Quit(5000); }
                    catch (Exception ex)
                    {
                        log.Error("Error closing Excel application during write", ex);
                    }
                }
            }
        }

        private static void WriteCsv(DataTable dt, string filePath)
        {
            StringBuilder sb = new StringBuilder();

            
            List<string> columnNames = new List<string>();
            foreach (DataColumn col in dt.Columns)
            {
                columnNames.Add(EscapeCsvField(col.ColumnName));
            }
            sb.AppendLine(string.Join(",", columnNames));

            
            foreach (DataRow row in dt.Rows)
            {
                List<string> fields = new List<string>();
                foreach (object item in row.ItemArray)
                {
                    fields.Add(item == DBNull.Value ? "" : EscapeCsvField(item.ToString()));
                }
                sb.AppendLine(string.Join(",", fields));
            }

            File.WriteAllText(filePath, sb.ToString(), Encoding.UTF8);
        }

        private static string EscapeCsvField(String field)
        {
            if (field.Contains(",") || field.Contains("\"") || field.Contains("\n") || field.Contains("\r"))
                return "\"" + field.Replace("\"", "\"\"") + "\"";
            return field;
        }

        private static void Log(string logFile, string message)
        {
            string entry = DateTime.Now.ToString("yyyy-MM-dd HH:mm:ss") + " - " + message;
            File.AppendAllText(logFile, entry + Environment.NewLine);
        }

        #endregion
    }

        private List<Dictionary<string, string>> ParseXlsbFile(string filePath)
        {
            List<Dictionary<string, string>> rows = new List<Dictionary<string, string>>();
            
            try
            {
                _xlh = new XLHelper(null);
                _xlh.CreateNewApp(false);
                
                if (_writer != null)
                    _writer("Opening XLSB file...");
                
                _xlh.OpenSheet(filePath, true);
                
                string worksheet = _xlh.ActiveWorksheetName();
                string usedRange = _xlh.UsedRangeAddressFull(worksheet);
                
                DataTable dt = _xlh.GetRangeDataTable(worksheet, usedRange);
                
                if (dt == null || dt.Rows.Count == 0)
                {
                    return rows;
                }
                
                
                foreach (DataRow dataRow in dt.Rows)
                {
                    Dictionary<string, string> row = new Dictionary<string, string>(StringComparer.OrdinalIgnoreCase);
                    
                    foreach (DataColumn col in dt.Columns)
                    {
                        string value = dataRow[col] != DBNull.Value ? dataRow[col]?.ToString()?.Trim() : null;
                        row[col.ColumnName] = value;
                    }
                    
                    rows.Add(row);
                }
            }
            finally
            {
                if (_xlh != null)
                {
                    try { _xlh.Quit(5000); }
                    catch (Exception ex)
                    {
                        log.Error("Error closing Excel application", ex);
                    }
                }
            }
            
            return rows;
        }

        private object GetValueOrNull(Dictionary<string, string> row, string key)
        {
            if (row.ContainsKey(key) && !string.IsNullOrWhiteSpace(row[key]))
            {
                return row[key].Trim();

        }

        private object GetAssetIdOrNull(Dictionary<string, string> row, string key)
        {
            if (row.ContainsKey(key) && !string.IsNullOrWhiteSpace(row[key]))
            {
                return ConvertAssetIdToString(row[key]);
            }
            return DBNull.Value;
        }




        private string ConvertAssetIdToString(string value)
        {
            if (string.IsNullOrWhiteSpace(value))
                return value;

            string trimmedValue = value.Trim();

            
            if (double.TryParse(trimmedValue, 
                System.Globalization.NumberStyles.Float | System.Globalization.NumberStyles.AllowExponent, 
                System.Globalization.CultureInfo.InvariantCulture, 
                out double doubleValue))
            {
                
                return doubleValue.ToString("F0", System.Globalization.CultureInfo.InvariantCulture);
            }

            
            return trimmedValue;
        }

        private object ParseDecimalOrNull(Dictionary<string, string> row, string key)
        {
            if (row.ContainsKey(key) && !string.IsNullOrWhiteSpace(row[key]))
            {
                string value = row[key].Trim().Replace("$", "").Replace(",", "");
                if (decimal.TryParse(value, out decimal result))
                {
                    return result;
                }
            }
            return DBNull.Value;
        }

        private object ParseDateOrNull(Dictionary<string, string> row, string key)
        {
            if (row.ContainsKey(key) && !string.IsNullOrWhiteSpace(row[key]))
            {
                string value = row[key].Trim();
                
                
                if (DateTime.TryParse(value, out DateTime result))
                {
                    return result.Date;
                }
                
                
                if (double.TryParse(value, out double oaDate))
                {
                    try
                    {
                        return DateTime.FromOADate(oaDate).Date;
                    }
                    catch
                    {
                        
                    }
                }
            }
            return DBNull.Value;
        }

        private void ValidateRowCompleteness(List<Dictionary<string, string>> rows)
        {
            if (rows.Count == 0)
                return;

            
            List<string> columns = new List<string>(rows[0].Keys);

            List<string> errors = new List<string>();

            foreach (var row in rows)
            {
                string rawAssetId = row.ContainsKey("Asset ID") ? row["Asset ID"] : null;
                if (string.IsNullOrWhiteSpace(rawAssetId))
                    continue;

                string assetId = ConvertAssetIdToString(rawAssetId);

                foreach (string col in columns)
                {
                    string value = row.ContainsKey(col) ? row[col] : null;
                    if (string.IsNullOrWhiteSpace(value))
                    {
                        errors.Add($"Asset ID {assetId}: missing value for column '{col}'");
                    }
                }
            }

            if (errors.Count > 0)
            {
                throw new InvalidOperationException(
                    "MISSING DATA ERROR: The following rows are missing required column values:\n" +
                    string.Join("\n", errors));
            }
        }

        private static void WithSftpSession(Action<Session> action)
        {
            using (var session = new Session())
            {
                string tempLog = Path.Combine(Path.GetTempPath(), "winscp_" + Guid.NewGuid().ToString("N") + ".log");
                session.SessionLogPath = tempLog;
                try
                {
                    session.Open(GetFASessionOptions());
                    action(session);
                }
                finally
                {
                    try { if (File.Exists(tempLog)) File.Delete(tempLog); } catch { }
                }
            }
        }

        internal static List<string> SftpListFilesMatching(string remoteDirectory, string containsPattern)
        {
            List<string> matchingFiles = new List<string>();
            WithSftpSession(session =>
            {
                RemoteDirectoryInfo dirInfo = session.ListDirectory(remoteDirectory);
                foreach (RemoteFileInfo fileInfo in dirInfo.Files)
                {
                    if (!fileInfo.IsDirectory && fileInfo.Name.Contains(containsPattern))
                        matchingFiles.Add(fileInfo.Name);
                }
            });
            return matchingFiles;
        }

namespace FirstAmericanLib
{









    public class ReconveyanceParser : DBLHelper.GenericBase, IDBLoader
    {
        private static readonly ILog log = LogManager.GetLogger(typeof(ReconveyanceParser));

        public ReconveyanceParser(string server) : base(server)
        {
        }

        public DBOutputHelper ParseAndLoad()
        {
            
            string inputDirectory = Database.Common.GetGlobal<string>(this._server, "BASE_FA_RECONVEYANCE_FOLDER", true);

            
            string[] pdfFiles = Directory.GetFiles(inputDirectory, "*.pdf");

            if (pdfFiles.Length == 0)
            {
                if (_writer != null)
                    _writer("No PDF files found in Reconveyance input directory.");
                return new DBOutputHelper { rows = 0 };
            }

            if (_writer != null)
                _writer($"Found {pdfFiles.Length} PDF files in Reconveyance directory.");

            
            Regex assetIdPattern = new Regex(@"^(\d{14})");

            
            Dictionary<string, List<string>> assetFiles = new Dictionary<string, List<string>>(StringComparer.OrdinalIgnoreCase);
            List<string> invalidFileNames = new List<string>();

            foreach (string pdfFile in pdfFiles)
            {
                string pdfFileName = Path.GetFileName(pdfFile);

                
                if (ReconveyanceLoads.CheckFileExists(this._server, pdfFileName))
                {
                    if (_writer != null)
                        _writer($"File {pdfFileName} has already been processed. Skipping...");
                    continue;
                }

                Match match = assetIdPattern.Match(pdfFileName);

                if (match.Success)
                {
                    string assetId = match.Groups[1].Value;
                    if (!assetFiles.ContainsKey(assetId))
                        assetFiles[assetId] = new List<string>();
                    assetFiles[assetId].Add(pdfFile);
                }
                else
                {
                    invalidFileNames.Add(pdfFileName);
                }
            }

            if (invalidFileNames.Count > 0)
            {
                if (_writer != null)
                    _writer($"WARNING: The following files do not begin with a 14-digit Asset ID and will be skipped: {string.Join(", ", invalidFileNames)}");
            }

            if (assetFiles.Count == 0)
            {
                if (_writer != null)
                    _writer("No new files to process.");
                return new DBOutputHelper { rows = 0 };
            }

            int assetSuccessCount = 0;
            int fileSuccessCount = 0;
            List<string> processedMaxIds = new List<string>();
            List<string> failedAssetIds = new List<string>();

            foreach (var kvp in assetFiles)
            {
                string maxId = kvp.Key;
                List<string> files = kvp.Value;

                try
                {
                    if (_writer != null)
                        _writer($"Processing Asset (Max ID): {maxId} ({files.Count} files)");

                    
                    string trueAssetId = BadgerLib.Assets.GetSingleAttributeSimple<string>(this._server, maxId, "asset_id", false) ?? maxId;

                    
                    string localFolder = BadgerLib.Assets.LocalFolderLocation(this._server, trueAssetId);

                    if (string.IsNullOrEmpty(localFolder))
                    {
                        throw new InvalidOperationException($"Could not resolve local folder for Asset ID: {maxId}");
                    }

                    
                    string finalDispositionFolder = Path.Combine(localFolder, "Final Disposition");

                    if (!Directory.Exists(finalDispositionFolder))
                    {
                        throw new DirectoryNotFoundException(
                            $"Final Disposition folder does not exist for Asset ID {maxId}: {finalDispositionFolder}");
                    }

                    
                    foreach (string file in files)
                    {
                        string fileName = Path.GetFileName(file);
                        string destPath = Path.Combine(finalDispositionFolder, fileName);

                        
                        if (File.Exists(destPath))
                            File.Delete(destPath);
                        File.Copy(file, destPath);

                        if (_writer != null)
                            _writer($"  Copied: {fileName} -> {finalDispositionFolder}");

                        
                        int loadId = ReconveyanceLoads.CreateNewLoadId(this._server, DateTime.Now, fileName, "Reconveyance", maxId);
                        ReconveyanceLoads.UpdateLatestFlag(this._server, loadId);

                        fileSuccessCount++;
                    }

                    
                    BadgerLib.Assets.WriteGenericAssetElement(this._server, "Settlements", "lien_release_completed", true, "asset_id", trueAssetId);

                    if (_writer != null)
                        _writer($"  Updated Settlements for true Asset ID: {trueAssetId}");

                    assetSuccessCount++;
                    processedMaxIds.Add(maxId);
                }
                catch (System.Exception ex)
                {
                    log.Error($"Error processing Reconveyance files for Asset ID {maxId}: {ex.Message}", ex);
                    failedAssetIds.Add(maxId);

                    if (_writer != null)
                        _writer($"  ERROR processing Asset ID {maxId}: {ex.Message}");
                }
            }

            
            DBOutputHelper oh = new DBOutputHelper();
            oh.rows = assetSuccessCount;
            _parms.Add("@success", assetSuccessCount);

            string body = $"First American Reconveyance Load Complete\n\n" +
                          $"Total Assets Processed: {assetSuccessCount}\n\n" +
                          $"Total Files Copied: {fileSuccessCount}\n\n" +
                          $"Max IDs: {string.Join(", ", processedMaxIds)}";

            if (failedAssetIds.Count > 0)
            {
                body += $"\n\nFailed Assets ({failedAssetIds.Count}): {string.Join(", ", failedAssetIds)}";
            }

            oh.email_body_override = body;

            if (_writer != null)
                _writer($"Reconveyance load complete. {assetSuccessCount} assets processed, {fileSuccessCount} files copied.");

            return oh;
        }
    }
